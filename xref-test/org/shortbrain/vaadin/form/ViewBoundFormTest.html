129     private void validate(ViewBoundForm form, int i) {
130         validateField(form, "aString", "A String", String.class, "str" + i, null);
131         validateField(form, "anInteger", "An Integer", String.class, ""+i, null);
132         validateField(form, "aDouble", "A Double", String.class, ""+(i + 0.0d), null);
133         validateField(form, "aValueOf", "A value of", Object.class, "valueOf" + i, propertyDataSource.get("values"));
134         validateField(form, "aList", "A list of value from", Set.class, Arrays.asList(new String[] { "0-" + i, "1-" + i, "2-" + i, "3-" + i, "4-" + i }), propertyDataSource.get("lists"));
135     }

137     private static void validateField(ViewBoundForm form, Object propertyId, String caption, Class<?> type, Object value, Container dataSource) {
138         Field f = form.getField(propertyId);
139         assertEquals(caption, f.getCaption());
140         assertEquals(type, f.getType());
141         if (f.getValue() instanceof Collection<?>) {
142             assertCollection((Collection) value, (Collection) f.getValue());
143         } else {
144             assertEquals(value, f.getValue());
145         }
146         if (dataSource != null && dataSource.size() > 0 && f instanceof Container.Viewer) {
147             Container.Viewer viewer = (Container.Viewer) f;
148             assertEquals(dataSource, viewer.getContainerDataSource());
149         }
150     }

152     private static void assertCollection(Collection<?> expected, Collection<?> actual) {
